Problem 1: Variable Scope & Global/Local

Understanding variable scope and the global keyword

Task:

Create a program that demonstrates the difference between global and local variables, and shows how to modify global variables inside functions.

Solution:

# Global variables
counter = 0
user_name = "Admin"

def modify_counter():
    global counter
    counter += 1
    print(f"Inside function: counter = {counter}")

def create_local_variable():
    # Local variable with same name as global
    counter = 100
    user_name = "Local User"
    print(f"Local counter: {counter}")
    print(f"Local user_name: {user_name}")

def access_global():
    # Accessing global variables without modification
    print(f"Global counter (read-only): {counter}")
    print(f"Global user_name (read-only): {user_name}")

# Demonstrate variable scope
print("=== Initial State ===")
print(f"Global counter: {counter}")
print(f"Global user_name: {user_name}")

print("\n=== Modifying Global Variable ===")
modify_counter()
modify_counter()
print(f"Global counter after modification: {counter}")

print("\n=== Local Variables ===")
create_local_variable()
print(f"Global counter unchanged: {counter}")
print(f"Global user_name unchanged: {user_name}")

print("\n=== Accessing Global Variables ===")
access_global()

# Common error demonstration
print("\n=== Error Handling ===")
def error_example():
    try:
        # This will cause UnboundLocalError if we try to modify
        # a global variable without declaring it global
        print(f"Trying to access: {counter}")
        # counter += 1  # Uncomment this line to see the error
    except UnboundLocalError as e:
        print(f"Error: {e}")

error_example()

Global user_name unchanged: Admin

=== Accessing Global Variables ===

Problem 2: Dynamic Variable Creation

Create variables dynamically using user input

Task:

Create a program that dynamically creates variables based on user input and stores them in a dictionary. Allow users to set, get, and list all stored variables.

Solution:

# Dynamic variable storage system
class VariableManager:
    def __init__(self):
        self.variables = {}
    
    def set_variable(self, name, value):
        """Set a variable with automatic type detection"""
        # Try to convert to appropriate type
        if value.isdigit():
            self.variables[name] = int(value)
        elif value.replace('.', '').isdigit():
            self.variables[name] = float(value)
        elif value.lower() in ['true', 'false']:
            self.variables[name] = value.lower() == 'true'
        else:
            self.variables[name] = value
        
        print(f"Variable '{name}' set to {self.variables[name]} (type: {type(self.variables[name]).__name__})")
    
    def get_variable(self, name):
        """Get a variable value"""
        if name in self.variables:
            return self.variables[name]
        else:
            return f"Variable '{name}' not found"
    
    def list_variables(self):
        """List all variables"""
        if not self.variables:
            print("No variables stored")
            return
        
        print("Stored Variables:")
        for name, value in self.variables.items():
            print(f"  {name} = {value} (type: {type(value).__name__})")
    
    def delete_variable(self, name):
        """Delete a variable"""
        if name in self.variables:
            del self.variables[name]
            print(f"Variable '{name}' deleted")
        else:
            print(f"Variable '{name}' not found")

# Demonstration
vm = VariableManager()

# Simulate user inputs
test_inputs = [
    ("name", "John Doe"),
    ("age", "25"),
    ("height", "5.9"),
    ("is_student", "true"),
    ("score", "87.5"),
    ("active", "false")
]

print("=== Setting Variables ===")
for var_name, var_value in test_inputs:
    vm.set_variable(var_name, var_value)

print("\n=== Listing All Variables ===")
vm.list_variables()

print("\n=== Getting Individual Variables ===")
for var_name, _ in test_inputs:
    result = vm.get_variable(var_name)
    print(f"{var_name}: {result}")

print("\n=== Testing Variable Operations ===")
# Calculate BMI using stored variables
age = vm.get_variable("age")
height = vm.get_variable("height")
print(f"Person is {age} years old and {height} feet tall")

print("\n=== Error Handling ===")
print(vm.get_variable("nonexistent_var"))

print("\n=== Deleting Variables ===")
vm.delete_variable("score")
vm.list_variables()

Key Concepts:

  • • Dynamic variable creation using dictionaries
  • • Automatic type detection and conversion
  • • Variable management (CRUD operations)
  • • Error handling for missing variables

Problem 3: Variable References & Memory

Understanding how Python handles variable references

Task:

Demonstrate the difference between mutable and immutable objects, and how variable assignment affects memory references.

Solution:

# Understanding variable references and memory
import copy

print("=== Immutable Objects (int, str, tuple) ===")
# Integers
a = 10
b = a
print(f"a = {a}, b = {b}")
print(f"id(a) = {id(a)}, id(b) = {id(b)}")
print(f"a is b: {a is b}")

# Modifying one variable
a = 20
print(f"\nAfter a = 20:")
print(f"a = {a}, b = {b}")
print(f"id(a) = {id(a)}, id(b) = {id(b)}")
print(f"a is b: {a is b}")

print("\n=== Mutable Objects (list, dict) ===")
# Lists
list1 = [1, 2, 3]
list2 = list1  # This creates a reference, not a copy!
print(f"list1 = {list1}, list2 = {list2}")
print(f"id(list1) = {id(list1)}, id(list2) = {id(list2)}")
print(f"list1 is list2: {list1 is list2}")

# Modifying one list affects both
list1.append(4)
print(f"\nAfter list1.append(4):")
print(f"list1 = {list1}, list2 = {list2}")
print("Both lists changed because they reference the same object!")

print("\n=== Copying Solutions ===")
# Shallow copy
original = [1, 2, [3, 4]]
shallow = copy.copy(original)
deep = copy.deepcopy(original)

print(f"Original: {original}")
print(f"Shallow copy: {shallow}")
print(f"Deep copy: {deep}")

# Modify nested list
original[2].append(5)
print(f"\nAfter modifying nested list in original:")
print(f"Original: {original}")
print(f"Shallow copy: {shallow}")  # Affected because nested list is shared
print(f"Deep copy: {deep}")        # Unaffected

print("\n=== Dictionary References ===")
person1 = {"name": "Alice", "age": 25}
person2 = person1
person3 = person1.copy()  # Shallow copy

print(f"person1: {person1}")
print(f"person2: {person2}")
print(f"person3: {person3}")

person1["age"] = 26
print(f"\nAfter person1['age'] = 26:")
print(f"person1: {person1}")
print(f"person2: {person2}")  # Changed (reference)
print(f"person3: {person3}")  # Unchanged (copy)

print("\n=== Practical Applications ===")
def modify_list(lst):
    """Function that modifies the original list"""
    lst.append("modified")
    return lst

def create_new_list(lst):
    """Function that creates a new list"""
    new_lst = lst.copy()
    new_lst.append("new")
    return new_lst

test_list = [1, 2, 3]
print(f"Original list: {test_list}")

result1 = modify_list(test_list)
print(f"After modify_list: {test_list}")  # Original is modified

result2 = create_new_list(test_list)
print(f"After create_new_list: {test_list}")  # Original unchanged
print(f"New list: {result2}")

🚨 Common Pitfalls:

  • • Assigning mutable objects creates references, not copies
  • • Modifying one reference affects all references to the same object
  • • Use .copy() for shallow copies, copy.deepcopy() for deep copies
  • • Functions can modify mutable arguments passed to them
